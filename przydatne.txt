usuniecie tabel

DO $$
DECLARE
    r RECORD;
BEGIN
    -- Pętla przez wszystkie tabele w schemacie 'public'
    FOR r IN (SELECT tablename FROM pg_tables WHERE schemaname = 'public') LOOP
        -- Dynamiczne wykonanie polecenia DROP TABLE
        -- Użycie 'CASCADE' jest kluczowe, aby usunąć tabele
        -- posiadające zależności (np. klucze obce)
        EXECUTE 'DROP TABLE IF EXISTS public.' || quote_ident(r.tablename) || ' CASCADE';
    END LOOP;
END;
$$;
